public function index(Request $request)
{
   $request->validate([
      'search' => ['nullable', 'string', 'max:255'],
      'kategori' => ['nullable', 'integer', 'exists:kategori,id'],
   ]);

   $query = Produk::with('kategori')->orderBy('id', 'desc');

   if ($request->filled('search')) {
      $query->where('nama', 'like', '%' . $request->search . '%');
   }

   if ($request->filled('kategori')) {
      $query->where('kategori_id', (int) $request->kategori);
   }

   $produkPage = $query->paginate(8)->withQueryString();
   $kategoriProduk = $produkPage->getCollection()->groupBy(function ($item) {
      return $item->kategori_id ?? 'uncategorized';
   });

   $kategoris = Kategori::orderBy('nama')->get();

   return view('produk', compact('kategoriProduk', 'produkPage', 'kategoris'));
}

<script>
const produkGrid = document.getElementById('produk-grid');
const loadMoreBtn = document.getElementById('load-more-btn');
const endCaption = document.getElementById('end-caption');

if (loadMoreBtn) {
   loadMoreBtn.addEventListener('click', async function () {
      const nextUrl = loadMoreBtn.dataset.nextUrl;
      if (!nextUrl) {
         loadMoreBtn.remove();
         endCaption.classList.remove('hidden');
         return;
      }

      const response = await fetch(nextUrl, {
         headers: { 'X-Requested-With': 'XMLHttpRequest' },
      });

      const html = await response.text();
      const parser = new DOMParser();
      const doc = parser.parseFromString(html, 'text/html');
      const nextItems = doc.querySelectorAll('#produk-grid .produk-item');

      nextItems.forEach((item) => {
         produkGrid.insertAdjacentHTML('beforeend', item.outerHTML);
      });

      const nextButtonOnFetchedPage = doc.getElementById('load-more-btn');
      if (nextButtonOnFetchedPage && nextButtonOnFetchedPage.dataset.nextUrl) {
         loadMoreBtn.dataset.nextUrl = nextButtonOnFetchedPage.dataset.nextUrl;
      } else {
         loadMoreBtn.remove();
         endCaption.classList.remove('hidden');
      }
   });
}
</script>

Schema::create('produk_ukuran_stok', function (Blueprint $table) {
   $table->id();
   $table->foreignId('produk_id')->constrained('produk')->cascadeOnDelete();
   $table->foreignId('ukuran_id')->constrained('ukuran')->cascadeOnDelete();
   $table->integer('stok')->default(0);
   $table->timestamps();

   $table->unique(['produk_id', 'ukuran_id']);
});
